# MailLife 项目架构与技术点详解

## 一、项目概述

MailLife 是一个 AI 驱动的邮件管理助手，通过 IMAP 自动同步邮件、用 DeepSeek AI 解析提取事件、LangGraph 多 Agent 编排生成摘要/通知，将邮件转化为可执行的时间线和日程。

## 六、IMAP 邮件同步（imap-tools 实现）

### 6.1 连接与认证

```python
from imap_tools import MailBox, AND
import ssl

class IMAPClient:
    def connect(self) -> bool:
        # 1. 创建 SSL 上下文（验证证书）
        ssl_context = ssl.create_default_context()

        # 2. 连接 QQ Mail IMAP 服务器
        self.mailbox = MailBox(
            self.host,    # imap.qq.com
            self.port,    # 993
            timeout=30,
            ssl_context=ssl_context
        )

        # 3. 使用邮箱+授权码登录
        self.mailbox.login(self.email, self.password)

        # 4. SSL 证书验证失败时，切换到跳过验证模式
        except ssl.SSLCertVerificationError:
            ssl_context.verify_mode = ssl.CERT_NONE  # 跳过验证
            self.mailbox.login(...)
```

### 6.2 邮件拉取策略

QQ Mail 的 IMAP 服务器对带日期条件的搜索（`SINCE`）响应极慢，因此采用**先获取后过滤**策略：

```python
def fetch_recent_emails(self, days=None, limit=100):
    # 1. 直接获取最新 N 封邮件（不带日期过滤，避免服务器超时）
    all_messages = list(self.mailbox.fetch(
        AND(all=True),    # 获取全部
        limit=limit,
        reverse=True      # 按时间倒序（最新在前）
    ))

    # 2. 客户端北京时区过滤
    beijing_tz = timezone(timedelta(hours=8))
    since_date = datetime.now(beijing_tz) - timedelta(days=days)

    for msg in all_messages:
        # 规范化邮件时间为 UTC+8
        msg_date_aware = msg.date.replace(tzinfo=beijing_tz)

        # 跳过超期邮件
        if since_date and msg_date_aware < since_date:
            continue

        # 提取邮件数据
        email_data = self._extract_email_data(msg)
        emails.append(email_data)
```

### 6.3 邮件正文提取

```python
def _get_email_body(self, msg) -> tuple:
    # 方法1：使用 imap_tools 的便捷属性
    text_content = msg.text   # 纯文本
    html_content = msg.html   # HTML

    # 方法2：手动遍历 MIME 结构（备选）
    if msg.is_multipart():
        for part in msg.walk():
            content_type = part.get_content_type()
            if content_type == "text/plain":
                payload = part.get_payload(decode=True)
                charset = part.get_content_charset() or 'utf-8'
                text_content = payload.decode(charset, errors='ignore')
```

```python
def sync_emails_to_db(days=7, limit=100) -> Dict:

    client.connect()
```

```python
def _build_prompt(self, subject: str, content: str) -> str:

    current_year = datetime.now().year

    return f"""你是一个智能邮件分析助手。请分析以下邮件内容，提取关键信息。

## 邮件信息
- 主题: {subject}
- 内容: {content[:2000]}   # 截断到2000字符
- 当前年份: {current_year}

    "event": {{

        "description": "事件描述",
        "event_time": "ISO格式时间，无法推断则填 null",
```

### 7.2 DeepSeek API 调用

```python
from openai import OpenAI

class EmailParser:
    def __init__(self):
        self.client = OpenAI(
            api_key=settings.OPENAI_API_KEY,
            base_url=settings.OPENAI_BASE_URL  # https://api.deepseek.com/v1
        )
        self.model = settings.OPENAI_MODEL    # deepseek-chat

    def parse_email(self, subject: str, content: str) -> Dict:
        prompt = self._build_prompt(subject, content)

        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": "你是一个智能邮件分析助手，擅长提取关键信息。"},
                {"role": "user", "content": prompt}
            ],
            temperature=0.3,    # 低随机性，保证解析稳定
            max_tokens=1000
        )

        result_text = response.choices[0].message.content.strip()

        # 从响应中提取 JSON（处理 ```json ... ``` 包裹格式）
        json_match = re.search(r'\{.*\}', result_text, re.DOTALL)
        if json_match:
            return json.loads(json_match.group())

        # 解析失败时使用关键词兜底
        return self._fallback_parse(subject, content)
```

### 7.3 解析结果保存

```python
def parse_and_save(self, email_id: int) -> bool:
    db = SessionLocal()
    try:
        email_record = db.query(Email).filter(Email.id == email_id).first()

        # 调用 AI 解析
        result = self.parse_email(email_record.subject, email_record.content_text)

        # 更新邮件记录
        email_record.category = result.get("category", "其他")
        email_record.summary = result.get("summary", "")
        email_record.processed = True

        # 创建事件记录（防重复）
        existing_event = db.query(Event).filter(
            Event.email_id == email_id
        ).first()

        if not existing_event:
            event_data = result.get("event", {})
            title = event_data.get("title") if event_data.get("title") else email_record.subject

            event = Event(
                email_id=email_id,
                event_type=result.get("category", "其他"),
                title=title,
                description=event_data.get("description", ""),
                event_time=self._parse_datetime(event_data.get("event_time"))
                       or email_record.received_at,  # 兜底：用邮件时间
                location=event_data.get("location") or None,
                important=result.get("important", False),
                actionable=result.get("actionable", False),
                action_items=json.dumps(result.get("action_items", [])),
                processed=True
            )
            db.add(event)

        db.commit()
        return True

    except Exception as e:
        db.rollback()
        return False
    finally:
        db.close()
```

### 7.4 时间解析（支持14种格式）

```python
def _parse_datetime(self, time_str: Optional[str]) -> Optional[datetime]:
    formats = [
        "%Y-%m-%dT%H:%M:%S",
        "%Y-%m-%dT%H:%M:%S.%f",
        "%Y-%m-%d %H:%M:%S",
        "%Y-%m-%d",
        "%m月%d日 %H:%M",
        "%m月%d日",
        "%Y年%m月%d日",
        # ... 共14种格式
    ]

    for fmt in formats:
        try:
            return datetime.strptime(time_str, fmt)
        except ValueError:
            continue

    # ISO 格式兜底
    try:
        return datetime.fromisoformat(time_str.replace('Z', '+00:00'))
    except:
        pass

    # 正则提取 YYYY-MM-DD
    match = re.search(r'(\d{4})-(\d{1,2})-(\d{1,2})', time_str)
    if match:
        return datetime(int(match.group(1)), int(match.group(2)), int(match.group(3)))

    return None
```

## 九、AI Agent 系统

### 9.1 两条技术路线

#### 路线 A：ReAct 引擎

**原理**：LLM 生成 `{thought, action, action_input}` → 执行工具 → 观察结果 → 循环

```python
# react_engine.py
class ReActEngine:
    def chat(self, message, conversation_history=None):
        messages = [
            {"role": "system", "content": self._build_system_prompt()}
        ]
        messages.extend(conversation_history or [])
        messages.append({"role": "user", "content": message})

        for iteration in range(self.max_iterations):  # 最多10次
            response = client.chat.completions.create(
                model=settings.OPENAI_MODEL,
                messages=messages,
                temperature=0.7
            )

            parsed = self._parse_response(response.content)
            thought = parsed["thought"]
            action = parsed["action"]
            action_input = parsed["action_input"]

            if action is None:
                # 无需执行工具，直接返回
                return AgentResult(success=True, data={"response": thought})

            # 执行工具
            tool_result = self.registry.execute(action, **action_input)

            # 将结果注入消息历史，LLM 继续思考
            messages.append({"role": "assistant", "content": response.content})
            messages.append({
                "role": "user",
                "content": f"工具执行结果：\n{tool_result.raw_output}\n请继续。"
            })

        return AgentResult(error=f"达到最大迭代次数 {self.max_iterations}")
```

```python
# email_agent.py
def fan_out_to_sub_agents(state: EmailAgentState) -> List[Send]:
    """返回 List[Send] 实现真正的并行执行"""
    intents = state.get("intents", [])

    node_map = {
        "parser":       "parser_agent",
        "summarizer":   "summarizer_agent",
        "notification": "notification_agent",
        "query":        "query_agent",
        "general":      "general_agent",
    }

    agents_to_run = []
    seen = set()
    for intent in intents:
        node_name = node_map.get(intent)
        if node_name and node_name not in seen:
            agents_to_run.append(Send(node_name, state))  # 并行触发
            seen.add(node_name)

    if not agents_to_run:
        agents_to_run.append(Send("general_agent", state))

    return agents_to_run
```

```python
                "type": "array",
                "items": {"type": "string", "enum": ["parser", "summarizer", "notification", "query", "general"]},
                "description": "意图列表，支持多意图并行",
                "minItems": 1, "maxItems": 5
            },
            "reasoning": {"type": "string", "description": "判断理由"},
            "params": {
                "type": "object",
                "properties": {
                    "limit": {"type": "integer"},
                    "action": {"type": "string"},
                    "query_type": {"type": "string", "enum": ["all", "emails", "events"]}
                }
            }
        },
        "required": ["intents", "reasoning"]
    }
}

def classify_intent_node(state: EmailAgentState) -> Dict:
    llm = _get_llm(temperature=0.1).bind_tools(
        [INTENT_SCHEMA], tool_choice="classify_intent"
    )

    response = llm.invoke([
        SystemMessage(content="你是意图分类器..."),
        HumanMessage(content=user_message)
    ])

    # 解析 tool_calls
    tool_call = response.tool_calls[0]
    args = json.loads(tool_call["arguments"])

    return {
        "intents": args["intents"],
        "intent_reasoning": args["reasoning"],
        "action_params": args.get("params", {})
    }
```

### 9.4 工具注册表实现

```python
# tools.py
class Tool(ABC):
    def __init__(self, name, description, parameters, func):
        self.name = name
        self.description = description
        self.parameters = parameters  # OpenAI Function Calling 格式
        self.func = func              # 实际执行函数

    def execute(self, **kwargs) -> ToolResult:
        try:
            result = self.func(**kwargs)
            return ToolResult(
                success=True,
                data=result,
                raw_output=json.dumps(result, ensure_ascii=False, default=str)
            )
        except Exception as e:
            return ToolResult(success=False, error=str(e))

    def get_schema(self) -> Dict:
        """返回 OpenAI Function Calling 格式"""
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.parameters
            }
        }


class ToolRegistry:
    def __init__(self):
        self._tools: Dict[str, Tool] = {}

    def register(self, tool: Tool):
        self._tools[tool.name] = tool

    def execute(self, name: str, **kwargs) -> ToolResult:
        tool = self._tools.get(name)
        if not tool:
            return ToolResult(success=False, error=f"Tool '{name}' not found")
        return tool.execute(**kwargs)

    def get_schemas(self) -> List[Dict]:
        return [tool.get_schema() for tool in self._tools.values()]
```

```python
        summarized = [m for m in all_messages if m.memory_type == "summarized"]
```
